CLI & GUI Version of run.py works with arguments

python3 run.py \
--text "Hello world" \
--split 200 \
--out ./output \
--font-color "#ffffff" \
--bg "#000000" \
--font-size 28 \
--padding 30


python3 run.py \
--text "Big text block here..." \
--split 250 \
--bg "#000000" \
--font-color "#ffffff" \
--border-color "#00ffff" \
--border 6 \
--out ./output

--text "Hello world" \
--split 200 \
--out ./output \
